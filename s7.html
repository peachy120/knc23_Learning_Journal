Session 7 Tutorial Questions

2. Select the element with the id elem1 and log its content to the console. Do this twice, first using querySelector() then using getElementById(). Notice the difference in specifying the element ID in these two versions.

Answers :

var a = document.querySelector("#elem1");
console.log(a.textContent);

var b = document.getElementById("elem1");
console.log(b.textContent);

3. Practice your selector skills: Select the following elements then (a) loop over them and log to the console their content, and (b) log to the console the number of elements:

a. all paragraph p elements

Answers :

var c = document.getElementsByTagName("p");
for(var i = 0; i < c.length; i++) {
    console.log(c[i].textContent);
}

var d = document.querySelectorAll("p");
for(var i = 0; i < d.length; i++) {
    console.log(d[i].textContent);
}

b. all elements with the class highlight

Answers :

var e = document.getElementsByClassName("highlight");
for(var i = 0; i < e.length; i++) {
    console.log(e[i].textContent);
}

var f = document.querySelectorAll(".highlight");
for(var i = 0; i < f.length; i++) {
    console.log(f[i].textContent);
}

c. all list item li elements in the list with id first

Answers :

var g = document.querySelectorAll("#first li");
for(var i = 0; i < g.length; i++) {
    console.log(g[i].textContent);
}

d. all list item li elements with the class highlight

Answers :

var h = document.querySelectorAll("li.highlight");
for(var i = 0; i < h.length; i++) {
    console.log(h[i].textContent);
}

e. all list item li elements and all paragraph p elements with the class highlight

Answers :

var k = document.querySelectorAll("li.highlight, p.highlight");
for(var i = 0; i < k.length; i++) {
    console.log(k[i].textContent);
}

6. The DOM provides node properties to access parent, sibling and child nodes. To test this, select the element with id first and then log to the console:

var n = document.querySelector("#first");

a. the node type, node name and text content of its first child node (.firstChild)

Answers :

var o = n.firstChild;
console.log("Node Type : ", o.nodeType);
console.log("Node Name : ", o.nodeName);
console.log("Text Content : ", o.textContent);

b. the node type, node name and text content of its next sibling node (.nextSibling)

Answers :

var p = n.nextSibling;
console.log("Node Type : ", p.nodeType);
console.log("Node Name : ", p.nodeName);
console.log("Text Content : ", p.textContent);

c. the node type and node name of its parent node (.parent)

Answers :

var q = n.parentNode;
console.log("Node Type : ", q.nodeType);
console.log("Node Name : ", q.nodeName);
console.log("Text Content : ", q.textContent);
